Usage

// ── v2 — named options, chainable instance ──────────────────
const chart = new ChutyChart({
  container:   'chart',          // DOM id
  data:        ohlcvArray,          // see formats below
  height:      400,
  candleWidth: 8,
  interpolate: false,
  priceAreaRatio:            0.8,  // 80% price, 20% volume
  volumeThresholdMultiplier: 2,   // outlier coloring
});
chart.render();

// Live update
chart.update(newData);

// ── Accepted data formats ────────────────────────────────────
// Short form (v1-compatible)
{ d: 1700000000, o: 100, h: 105, l: 98, c: 103, v: 1200000 }

// Verbose form (new)
{ timestamp: 1700000000000, open: 100, high: 105,
  low: 98, close: 103, volume: 1200000 }

// Millisecond timestamps auto-converted ✓
